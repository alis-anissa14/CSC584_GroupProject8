CREATE TABLE FEEDBACK (FEEDBACK_ID INTEGER DEFAULT AUTOINCREMENT: start 1 increment 1  NOT NULL GENERATED ALWAYS AS IDENTITY, SUBMISSION_ID INTEGER, SUPERVISOR_ID INTEGER, FEEDBACK_FILE_PATH VARCHAR(255), FEEDBACK_TEXT VARCHAR(255), FEEDBACK_DATE TIMESTAMP DEFAULT CURRENT_TIMESTAMP , PRIMARY KEY (FEEDBACK_ID));

CREATE TABLE MILESTONE (MILESTONE_ID INTEGER DEFAULT AUTOINCREMENT: start 1 increment 1  NOT NULL GENERATED ALWAYS AS IDENTITY, PROJECT_ID INTEGER, TITLE VARCHAR(100), TASK VARCHAR(255), START_DATE TIMESTAMP, END_DATE TIMESTAMP, STATUS VARCHAR(20), PRIMARY KEY (MILESTONE_ID));

CREATE TABLE PROJECT (PROJECT_ID INTEGER DEFAULT AUTOINCREMENT: start 1 increment 1  NOT NULL GENERATED ALWAYS AS IDENTITY, TITLE VARCHAR(150) NOT NULL, STUDENT_ID INTEGER, SUPERVISOR_ID INTEGER, PROJECT_STATUS VARCHAR(20), PRIMARY KEY (PROJECT_ID));

CREATE TABLE SUBMISSION (SUBMISSION_ID INTEGER DEFAULT AUTOINCREMENT: start 1 increment 1  NOT NULL GENERATED ALWAYS AS IDENTITY, MILESTONE_ID INTEGER, SUBMISSION_DATE TIMESTAMP DEFAULT CURRENT_TIMESTAMP , SUBMISSION_FILE_PATH VARCHAR(255), SUBMISSION_REMARKS VARCHAR(255), SUBMISSION_STATUS VARCHAR(20), PRIMARY KEY (SUBMISSION_ID));

CREATE TABLE USERS (USER_ID INTEGER NOT NULL, EMAIL VARCHAR(100), PASSWORD VARCHAR(100), FULL_NAME VARCHAR(100), "ROLE" VARCHAR(20), PROGRAM_CODE VARCHAR(20), SESSIONS VARCHAR(10), PRIMARY KEY (USER_ID));
